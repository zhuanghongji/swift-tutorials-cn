public enum NotificationKey : String {
    // ...

    // 指示 SwiftData 正在使用哪一代模型存储的令牌
    case queryGeneration

    // 一组标识上下文无效模型的值
    case invalidatedAllIdentifiers

    // 一组标识上下文插入模型的值
    case insertedIdentifiers

    // 一组标识上下文更新模型的值
    case updatedIdentifiers

    // 一组标识上下文已删除模型的值
    case deletedIdentifiers

    // Creates a new instance with the specified raw value.
    //
    // If there is no value of the type that corresponds with the specified raw
    // value, this initializer returns `nil`. For example:
    //
    //     enum PaperSize: String {
    //         case A4, A5, Letter, Legal
    //     }
    //
    //     print(PaperSize(rawValue: "Legal"))
    //     // Prints "Optional("PaperSize.Legal")"
    //
    //     print(PaperSize(rawValue: "Tabloid"))
    //     // Prints "nil"
    //
    // - Parameter rawValue: The raw value to use for the new instance.


    // The raw type that can be used to represent all values of the conforming
    // type.
    //
    // Every distinct value of the conforming type has a corresponding unique
    // value of the `RawValue` type, but there may be values of the `RawValue`
    // type that don't have a corresponding value of the conforming type.
    public typealias RawValue = String

    // The corresponding value of the raw type.
    //
    // A new instance initialized with `rawValue` will be equivalent to this
    // instance. For example:
    //
    //     enum PaperSize: String {
    //         case A4, A5, Letter, Legal
    //     }
    //
    //     let selectedSize = PaperSize.Letter
    //     print(selectedSize.rawValue)
    //     // Prints "Letter"
    //
    //     print(selectedSize == PaperSize(rawValue: selectedSize.rawValue)!)
    //     // Prints "true"
    public var rawValue: String { get }
}

创建通知​

init(rawValue:)​
// 使用指定的原始值创建一个新实例
//
// - rawValue: 用于新实例的原始值
public init?(rawValue: String)

如果没有与指定原始值对应的类型值，则此初始值设定项返回 nil 。例如：

enum PaperSize: String {
    case A4, A5, Letter, Legal
}

print(PaperSize(rawValue: "Legal"))
// Prints "Optional("PaperSize.Legal")"

print(PaperSize(rawValue: "Tabloid"))
// Prints "nil"

实例属性​

rawValue​
// 原始类型对应的值
public var rawValue: String { get }

用 rawValue 初始化的新实例将与此实例等效。例如：

enum PaperSize: String {
    case A4, A5, Letter, Legal
}

let selectedSize = PaperSize.Letter
print(selectedSize.rawValue)
// Prints "Letter"

print(selectedSize == PaperSize(rawValue: selectedSize.rawValue)!)
// Prints "true"

类型别名​

RawValue​
// 可用于表示符合类型的所有值的原始类型
public typealias RawValue = String

一致类型的每个不同值都有一个对应的 RawValue 类型的唯一值，但可能有 RawValue 类型的值没有一致类型的对应值。
