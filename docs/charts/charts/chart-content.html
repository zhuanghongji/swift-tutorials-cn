ChartContent

原文：ChartContent

表示你在图表上绘制的内容的类型。

@available(iOS 16.0, macOS 13.0, tvOS 16.0, watchOS 9.0, *)
public protocol ChartContent {

    // 此实例主体中包含的图表内容的类型
    associatedtype Body : ChartContent

    // 图表内容的内容和行为
    @ChartContentBuilder var body: Self.Body { get }
}

@available(iOS 16.0, macOS 13.0, tvOS 16.0, watchOS 9.0, *)
extension ChartContent {
    // ...
}

概述​

你可以通过将符合 ChartContent 协议的实例添加到图表的 content 闭包来构建 Chart。以下示例将三个显式 BarMark 实例添加到图表中：

Chart {
    BarMark(
        x: .value("Shape Type", "Cube"),
        y: .value("Total Count", 5)
    )
    BarMark(
        x: .value("Shape Type", "Sphere"),
        y: .value("Total Count", 6)
    )
    BarMark(
        x: .value("Shape Type", "Pyramid"),
        y: .value("Total Count", 4)
    )
}

图表绘制与你指定的实例相对应的标记：

你可以将任意数量或任意类型的标记组合在一个图表中，就如上例所示那样单独列出它们，将它们包装在 ForEach 中，或者这些的任意组合。对于某些标记类型，例如 LineMark，你可以使用标记的 series 初始化参数将标记分组为系列。

Chart {
    ForEach(1...3, id: \.self) { num in
        LineMark(
            x: .value("Day", "\(num)"),
            y: .value("Count", num * 3),
            series: .value("Fruit", "Apple")
        )
        .foregroundStyle(.red)
    }

    ForEach(1...3, id: \.self) { num in
        LineMark(
            x: .value("Day", "\(num)"),
            y: .value("Count", num * 2),
            series: .value("Fruit", "Orange")
        )
        .foregroundStyle(.orange)
    }

    // 尽管在位置上隔了个 Orange 系列
    // 但该标记跟其指定的那样，是属于 Apple 系列的
    LineMark(
        x: .value("Day", "4"),
        y: .value("Count", 18),
        series: .value("Fruit", "Apple")
    )
}
.padding()

配置图表内容​

ChartContent 协议提供了一组修饰器，可用于配置图表内容的属性。它们的行为类似于 SwiftUI 视图修饰器，只不过它们作用于图表内容而不是视图。任何符合该协议的类型都可以使用这些修饰器。例如，你可以将 foregroundStyle(_:) 修饰器添加到上一示例中表示球体（spheres）数量的条形图上，使其变为红色：

BarMark(
    x: .value("Shape Type", "Sphere"),
    y: .value("Total Count", 6)
)
.foregroundStyle(.red)

lineStyle(_:)​

// 设置线标记的样式
//
// - style: 笔画风格
public func lineStyle(_ style: StrokeStyle) -> some ChartContent

示例如下：

let nums = [1, 3, 5, 4, 6]

var body: some View {
    Chart {
        ForEach(nums, id: \.self) { num in
            LineMark(
                x: .value("Day", "\(num)"),
                y: .value("Count", num * 3),
                series: .value("Type", "A")
            )
            .lineStyle(.init(lineWidth: 30, lineCap: .round))
        }

        ForEach(nums, id: \.self) { num in
            LineMark(
                x: .value("Day", "\(num)"),
                y: .value("Count", num),
                series: .value("Type", "B")
            )
            .foregroundStyle(.red)
            .lineStyle(.init(lineWidth: 20, lineCap: .square, lineJoin: .round))
        }
    }
}

注意

仅当你有预定义的 stroke 样式时才使用此 .lineStyle(_:) 重载。提供的描边样式将覆盖线标记的默认线宽和线帽。

interpolationMethod(_:)​

// 使用你指定的插值方法绘制线条和区域标记
//
// - method: 插值方法
public func interpolationMethod(_ method: InterpolationMethod) -> some ChartContent

示例如下：

let nums = [1, 3, 5, 4, 6]

var body: some View {
    Chart {
        ForEach(nums, id: \.self) { num in
            LineMark(
                x: .value("Day", "\(num)"),
                y: .value("Count", num * 3),
                series: .value("Type", "A")
            )
            .lineStyle(.init(lineWidth: 30, lineCap: .round))
            .interpolationMethod(.cardinal)
        }

        ForEach(nums, id: \.self) { num in
            LineMark(
                x: .value("Day", "\(num)"),
                y: .value("Count", num),
                series: .value("Type", "B")
            )
            .foregroundStyle(.red)
            .lineStyle(.init(lineWidth: 20, lineCap: .square, lineJoin: .round))
            .interpolationMethod(.monotone)
        }
    }
}

offset(x:y:)​

// 对图表内容应用垂直和水平偏移
//
// - x: 屏幕坐标中的水平偏移
// - y: 屏幕坐标中的垂直偏移
public func offset(x: CGFloat = 0, y: CGFloat = 0) -> some ChartContent

示例如下：

let nums = [1, 3, 5, 4, 6]

var body: some View {
    Chart {
        ForEach(nums, id: \.self) { num in
            LineMark(
                x: .value("Day", "\(num)"),
                y: .value("Count", num),
                series: .value("Type", "B")
            )
            .lineStyle(.init(lineWidth: 20, lineCap: .square, lineJoin: .round))
            .opacity(0.2)

            LineMark(
                x: .value("Day", "\(num)"),
                y: .value("Count", num),
                series: .value("Type", "C")
            )
            .lineStyle(.init(lineWidth: 20, lineCap: .square, lineJoin: .round))
            .offset(x: 20, y: 100)
        }
    }
    .padding()
}

offset(x:yStart:yEnd:)​

// 对图表内容应用偏移

// - yStart: 屏幕坐标中的起始垂直偏移
// - yEnd: 屏幕坐标中的结束垂直偏移
public func offset(x: CGFloat = 0,
                   yStart: CGFloat = 0,
                   yEnd: CGFloat = 0

yStart 和 yEnd 偏移值仅适用于具有该类属性的标记，例如条形标记和线段标记。

注释标记​

annotation(position:alignment:spacing:content:)​

// 使用相对于其边界定位的视图来注释此标记或标记集合
//
// - position: 相对于正在注释的项目的位置，注释将放置在该位置
// - alignment: 用于将注释对齐到指定位置的指南

) -> some ChartContent where C : View

annotation(position:alignment:spacing:content:)​

// 使用相对于其边界定位的视图来注释此标记或标记集合
