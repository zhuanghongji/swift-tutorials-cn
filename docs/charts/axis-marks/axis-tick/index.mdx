## 结构体

### Length

```swift
// 描述刻度的长度
public struct Length : CustomStringConvertible {

    // 自动确定刻度长度
    public static var automatic: AxisTick.Length { get }

    // 描述延伸至其关联标签的刻度
    public static var label: AxisTick.Length { get }

    // 描述延伸到轴上最长标签的刻度
    public static var longestLabel: AxisTick.Length { get }

    // 描述一个延伸到其关联标签的刻度，具有给定的附加长度
    //
    // - extendPastBy: 添加到关联标签后面的附加长度
    public static func label(extendPastBy: CGFloat = 0) -> AxisTick.Length

    // 描述延伸到轴上最长标签的刻度，具有给定的附加长度
    //
    // - extendPastBy: 添加到关联标签后面的附加长度
    public static func longestLabel(extendPastBy: CGFloat = 0) -> AxisTick.Length
}
```
